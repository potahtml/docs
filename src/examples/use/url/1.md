title: Sanitize and decode URLs safely
description: Trim copy-paste artefacts and decode without throwing.

`pota/use/url` exposes `cleanLink` (trim trailing punctuation that
clings to copied links), a forgiving `decodeURIComponent` that
returns the input on malformed sequences, and the protocol probes
`isFileProtocol` / `hasProtocol`. `removeNestedProtocol` strips
constructs like `blob:http://...`.

```jsx
import {
	cleanLink,
	decodeURIComponent,
	hasProtocol,
} from 'pota/use/url'
import { render } from 'pota'

const examples = [
	'https://example.com/post.',
	'https://example.com?q=hello%2Cworld',
	'https://example.com?bad=%',
	'/just/a/path',
]

function App() {
	return (
		<ul>
			{examples.map(raw => (
				<li>
					raw: <code>{raw}</code>
					<br />
					clean: <code>{cleanLink(raw)}</code>
					<br />
					decoded: <code>{decodeURIComponent(raw)}</code>
					<br />
					hasProtocol: {String(hasProtocol(raw))}
				</li>
			))}
		</ul>
	)
}

render(<App />)
```
