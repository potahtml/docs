---
title: Paginate from a server
description: Provide an async `fetch` so each page loads on demand.
category: use
api: paginate
import: pota/use/paginate
tags: [paginate, async, fetch]
---

`paginate(fetch, options)` accepts a `fetch(start, end)` that may
return a `Promise`. The `items` derived unwraps the promise — until
it resolves, `items()` returns `undefined`, which `<Suspense>` and
`isResolved(items)` both understand. Pass `numItems` as a memo so
the page count updates when the server reports a different total.

```jsx
import { paginate } from 'pota/use/paginate'
import { memo, render, signal } from 'pota'
import { For, Suspense } from 'pota/components'

const total = signal(100)

function App() {
	const numPerPage = signal(10)

	const page = paginate(async (start, end) => {
		await new Promise(r => setTimeout(r, 300))
		return Array.from(
			{ length: end - start },
			(_, i) => `row ${start + i + 1}`,
		)
	}, {
		numPerPage: numPerPage.read,
		numItems: memo(() => total.read()),
	})

	return (
		<div>
			<p>
				page {page.currentPage} / {page.totalPages}
			</p>
			<Suspense fallback={<p>loading…</p>}>
				<ul>
					<For each={page.items}>{x => <li>{x}</li>}</For>
				</ul>
			</Suspense>
			<button
				on:click={page.previous}
				prop:disabled={() => !page.hasPrevious()}
			>
				← prev
			</button>
			<button
				on:click={page.next}
				prop:disabled={() => !page.hasNext()}
			>
				next →
			</button>
		</div>
	)
}

render(<App />)
```
