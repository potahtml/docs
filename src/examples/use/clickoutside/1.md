---
title: Close a popover on outside click
description: Dismiss a UI element when the user clicks anywhere outside it.
category: use
api: clickOutside
import: pota/use/clickoutside
tags: [clickoutside, popover, use:ref]
---

`pota/use/clickoutside` exports a `clickOutside(handler, options?)`
factory. It returns a ref function: attach it via `use:ref` and the
handler runs whenever a `pointerdown` lands outside the element.
Pass `{ once: true }` for single-shot dismissal.

```jsx
import { render, signal } from 'pota'
import { Show } from 'pota/components'
import { clickOutside } from 'pota/use/clickoutside'

function App() {
	const open = signal(false)

	return (
		<div>
			<button on:click={() => open.write(true)}>open</button>
			<Show when={open.read}>
				<div use:ref={clickOutside(() => open.write(false))}>
					<p>I close on outside click.</p>
				</div>
			</Show>
		</div>
	)
}

render(<App />)
```
