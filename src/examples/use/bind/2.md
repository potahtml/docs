---
title: Bind across radio, select, textarea, and contenteditable
description: One `bind`-derived signal per field, regardless of input type.
category: use
api: bind
import: pota/use/bind
tags: [bind, radio, select, contenteditable]
---

The `use:bind` plugin adapts to the element type: `<input
type="radio">` writes when checked, `<select>` and `<textarea>`
mirror their respective values, and contenteditable elements track
`innerText` while preserving caret position. Same `bind()` call,
different DOM surfaces.

```jsx
import { bind } from 'pota/use/bind'
import { render } from 'pota'

function App() {
	const size = bind('m')
	const country = bind('US')
	const notes = bind('')
	const headline = bind('Edit me')

	return (
		<form>
			<fieldset>
				<legend>size</legend>
				<label>
					<input type="radio" name="size" value="s" use:bind={size} /> S
				</label>
				<label>
					<input type="radio" name="size" value="m" use:bind={size} /> M
				</label>
				<label>
					<input type="radio" name="size" value="l" use:bind={size} /> L
				</label>
			</fieldset>

			<label>
				country:{' '}
				<select use:bind={country}>
					<option value="US">United States</option>
					<option value="UY">Uruguay</option>
					<option value="DE">Germany</option>
				</select>
			</label>

			<label>
				notes:
				<textarea use:bind={notes} rows="3" />
			</label>

			<label>
				headline:
				<h2 contenteditable="plaintext-only" use:bind={headline} />
			</label>

			<pre>{() => JSON.stringify({ size: size(), country: country(), notes: notes(), headline: headline() }, null, 2)}</pre>
		</form>
	)
}

render(<App />)
```
