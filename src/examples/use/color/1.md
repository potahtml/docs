`pota/use/color` exposes a set of color utilities built on
[`color-bits`](https://www.npmjs.com/package/color-bits): pick a
`textColor` that has good contrast on a background (APCA), build a
perceptually-uniform `scale` of stops in OkLab, validate user input
with `validateColor`, or open the platform `eyeDropper`.

```jsx
import { eyeDropper, scale, textColor } from 'pota/use/color'
import { For } from 'pota/components'
import { render, signal } from 'pota'

function App() {
	const bg = signal('#3366ff')
	const stops = scale(['#ff0066', '#3366ff'], 6)

	return (
		<div>
			<div
				style={{
					background: bg.read,
					color: () => textColor(bg.read()),
					padding: '1rem',
				}}
			>
				background: {bg.read} (text auto-contrasts)
			</div>
			<button on:click={() => eyeDropper(c => bg.write(c))}>
				pick a color
			</button>
			<div style={{ display: 'flex' }}>
				<For each={stops}>
					{c => (
						<div
							style={{
								background: c,
								width: '40px',
								height: '40px',
							}}
						/>
					)}
				</For>
			</div>
		</div>
	)
}

render(<App />)
```
