---
title: Sortable list with swap
description: Reorder rows without remounting them — `For` keys by item identity.
category: components
api: For
import: pota/components
tags: [For, sortable, reorder]
---

When you swap two items in the array, `For` moves the existing DOM
nodes instead of recreating them — any state inside a row (input
focus, animation progress, a child component's signal) survives the
reorder. Pass `restoreFocus` so `For` even returns the focused
element to where it was after the swap.

```jsx
import { render, signal } from 'pota'
import { For } from 'pota/components'

function App() {
	const items = signal([
		{ id: 'a', text: 'apple' },
		{ id: 'b', text: 'banana' },
		{ id: 'c', text: 'cherry' },
	])

	function swap(i, j) {
		items.update(list => {
			const next = [...list]
			;[next[i], next[j]] = [next[j], next[i]]
			return next
		})
	}

	return (
		<ul>
			<For each={items.read} restoreFocus>
				{(item, i) => (
					<li>
						<input
							prop:value={item.text}
							on:input={e => (item.text = e.currentTarget.value)}
						/>
						<button
							on:click={() => i > 0 && swap(i, i - 1)}
						>
							↑
						</button>
						<button
							on:click={() =>
								i < items.read().length - 1 && swap(i, i + 1)
							}
						>
							↓
						</button>
					</li>
				)}
			</For>
		</ul>
	)
}

render(<App />)
```
