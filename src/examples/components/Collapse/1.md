---
title: Hide without unmounting
description: Toggle children's visibility while keeping their state alive.
category: components
api: Collapse
import: pota/components
tags: [Collapse, visibility]
---

`<Collapse when>` is `Show`'s sibling that uses
`style:display="none"` instead of removing the subtree. State inside
the children survives across hide/show — useful for tabs, accordions,
or expensive widgets you don't want to re-mount.

```jsx
import { render, signal } from 'pota'
import { Collapse } from 'pota/components'

function HeavyForm() {
	const draft = signal('')
	return (
		<form>
			<label>draft (preserved when hidden):</label>
			<input
				prop:value={draft.read}
				on:input={e => draft.write(e.currentTarget.value)}
			/>
			<p>{() => `${draft.read().length} chars`}</p>
		</form>
	)
}

function App() {
	const open = signal(true)

	return (
		<div>
			<button on:click={() => open.update(o => !o)}>
				toggle
			</button>
			<Collapse when={open.read}>
				<HeavyForm />
			</Collapse>
		</div>
	)
}

render(<App />)
```
