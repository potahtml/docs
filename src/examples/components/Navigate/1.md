---
title: Redirect on render
description: Navigate to a path while rendering its children once.
category: components
api: Navigate
import: pota/components
tags: [Navigate, routing, redirect]
---

`<Navigate path="...">` triggers a navigation as soon as it renders.
Useful as the body of a fallback `<Route>` for redirects, or inside
a `<Show>` to bounce the user when a condition is met. Optional
`replace` swaps history instead of pushing.

```jsx
import { render, signal } from 'pota'
import { Navigate, Route, Show } from 'pota/components'

function App() {
	const loggedIn = signal(false)

	return (
		<div>
			<button on:click={() => loggedIn.write(true)}>log in</button>

			<Route path="/protected">
				<Show
					when={loggedIn.read}
					fallback={<Navigate path="/login" replace />}
				>
					<p>secret content</p>
				</Show>
			</Route>

			<Route path="/login">
				<p>please log in</p>
			</Route>
		</div>
	)
}

render(<App />)
```
