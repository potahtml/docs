---
title: Pick the first matching branch
description: Render the first `<Match>` whose `when` is truthy, or a fallback.
category: components
api: Switch
import: pota/components
tags: [Switch, Match, branch]
---

`<Switch>` evaluates each child `<Match>` in order and renders the
first one whose `when` is truthy. The fallback is either an explicit
`fallback` prop or the last `<Match>` declared without a `when`.
Cleaner than nested `Show`s when you have several mutually-exclusive
states.

```jsx
import { render, signal } from 'pota'
import { Match, Switch } from 'pota/components'

function App() {
	const status = signal('loading')

	return (
		<div>
			<button on:click={() => status.write('loading')}>
				loading
			</button>
			<button on:click={() => status.write('success')}>
				success
			</button>
			<button on:click={() => status.write('error')}>
				error
			</button>

			<Switch fallback={<p>unknown</p>}>
				<Match when={() => status.read() === 'loading'}>
					<p>loading…</p>
				</Match>
				<Match when={() => status.read() === 'success'}>
					<p>done!</p>
				</Match>
				<Match when={() => status.read() === 'error'}>
					<p>oh no</p>
				</Match>
			</Switch>
		</div>
	)
}

render(<App />)
```
