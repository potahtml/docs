---
title: Nested routes and reactive params
description: Read `:id` and `:slug` from `location.params` inside child routes.
category: components
api: Route
import: pota/components
tags: [Route, nested, params]
---

Child `<Route>`s match relative to their parent's path: a
`<Route path="/users">` containing `<Route path=":id">` requires
`/users/42` and exposes `id` via `location.params`. The params
object is reactive — keys update as the URL changes, so JSX that
reads them stays in sync.

```jsx
import { render } from 'pota'
import { A, Route } from 'pota/components'
import { location } from 'pota/use/location'

function UserDetail() {
	const params = location.params
	return (
		<div>
			<h2>user #{() => params.id}</h2>
			<p>open the URL with a different id to see this update</p>
		</div>
	)
}

function App() {
	return (
		<div>
			<nav>
				<A href="/users/1">user 1</A>
				{' · '}
				<A href="/users/2">user 2</A>
				{' · '}
				<A href="/users/42/posts/intro">42 / posts / intro</A>
			</nav>

			<Route path="/users">
				<Route path=":id">
					<UserDetail />
					<Route path="posts/:slug">
						<p>post: {() => location.params.slug}</p>
					</Route>
				</Route>
			</Route>
		</div>
	)
}

render(<App />)
```
